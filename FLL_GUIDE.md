```python
def _decision_directed(self, sample):
    # –ñ—ë—Å—Ç–∫–æ–µ —Ä–µ—à–µ–Ω–∏–µ –¥–ª—è QPSK
```

2. –ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –∫–æ–º–ø–ª–µ–∫—Å–Ω—ã–π —ç–∫—Å–ø–æ–Ω–µ–Ω—Ü–∏–∞–ª –¥–ª—è –∫–æ—Ä—Ä–µ–∫—Ü–∏–∏

### –ü—Ä–∏–Ω—Ü–∏–ø —Ä–∞–±–æ—Ç—ã

NCO –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞–µ—Ç **—Ç–µ–∫—É—â—É—é —Ñ–∞–∑—É** –∏ **—Ç–µ–∫—É—â—É—é —á–∞—Å—Ç–æ—Ç—É**:


–í—ã—Ö–æ–¥–Ω–æ–π —Å–∏–≥–Ω–∞–ª `e^(-j¬∑œÜ)` **—É–¥–∞–ª—è–µ—Ç** —á–∞—Å—Ç–æ—Ç–Ω—ã–π —Å–¥–≤–∏–≥ –∏–∑ –≤—Ö–æ–¥–Ω–æ–≥–æ —Å–∏–≥–Ω–∞–ª–∞.

### –ö–æ–¥

```python
class NCO:
    def __init__(self, initial_freq=0.0):
        self.freq = initial_freq
        self.phase = 0.0
    
    def update_freq(self, freq_correction):
        self.freq += freq_correction
    
    def step(self):
        # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º –∫–æ—Ä—Ä–µ–∫—Ü–∏—é
        correction_signal = np.exp(-1j * self.phase)
        
        # –û–±–Ω–æ–≤–ª—è–µ–º —Ñ–∞–∑—É
        self.phase += 2 * np.pi * self.freq
        
        # Wrap —Ñ–∞–∑—ã –≤ [-œÄ, œÄ]
        self.phase = np.arctan2(np.sin(self.phase), np.cos(self.phase))
        
        return correction_signal
```

### Wrap —Ñ–∞–∑—ã

–í–∞–∂–Ω–æ –¥–µ—Ä–∂–∞—Ç—å —Ñ–∞–∑—É –≤ –ø—Ä–µ–¥–µ–ª–∞—Ö `[-œÄ, œÄ]`, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –ø–µ—Ä–µ–ø–æ–ª–Ω–µ–Ω–∏—è:

```python
while self.phase > np.pi:
```

```python
    def __init__(self, detector_method='cross_product', Kp=0.01, Ki=0.001):
        self.detector = FrequencyErrorDetector(method=detector_method)
        self.loop_filter = LoopFilter(Kp=Kp, Ki=Ki)
        self.nco = NCO()
    
    def process_sample(self, sample):
        # 1. –ö–æ—Ä—Ä–µ–∫—Ü–∏—è –≤—Ö–æ–¥–Ω–æ–≥–æ —Å–∏–≥–Ω–∞–ª–∞
        correction = self.nco.step()
        corrected_sample = sample * correction
        
        # 2. –î–µ—Ç–µ–∫—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –æ—à–∏–±–∫–∏
        error = self.detector.detect(corrected_sample)
        
        # 3. –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è
        freq_correction = self.loop_filter.update(error)
        
        # 4. –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ NCO
        self.nco.update_freq(freq_correction)
        
        return corrected_sample
```
